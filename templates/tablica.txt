        async function updateDashboard() {
            try {
                const response = await fetch('/status');
                if (!response.ok) throw new Error(`B≈ÇƒÖd sieciowy: ${response.status}`);
                const data = await response.json();

                document.getElementById('status-text').textContent = data.status_text;
                document.getElementById('balance-value').textContent = `${data.balance} ${data.quote_asset}`;
                
                const statusBox = document.getElementById('status-box');
                statusBox.className = 'status-box';
                statusBox.classList.add(data.in_position ? 'status-active' : 'status-idle');
                
            } catch (error) {
                document.getElementById('status-text').textContent = "B≈ÅƒÑD PO≈ÅƒÑCZENIA";
                console.error("B≈ÇƒÖd od≈õwie≈ºania:", error);
            }
        }

        function updateClock() {
            const now = new Date();
            const timeEl = document.querySelector('#clock .time');
            if(timeEl) timeEl.textContent = now.toLocaleTimeString('pl-PL');
        }

        document.addEventListener('DOMContentLoaded', () => {
            updateDashboard();
            setInterval(updateDashboard, 5000);
            setInterval(updateClock, 1000);
        });
    </script>
